        private void download_click(object sender, EventArgs e)
        {
            OpenFileDialog open_dialog = new OpenFileDialog(); 
            if (open_dialog.ShowDialog() == DialogResult.OK) 
            {
                try
                {
                    image = new Bitmap(open_dialog.FileName);
                    pictureBox1.Image = image;
                }
                catch
                {
                    DialogResult rezult = MessageBox.Show("Невозможно открыть выбранный файл","Ошибка", MessageBoxButtons.OK, MessageBoxIcon.Error);//разобраться
                }
            }
        }

        {

        }


        private void inversion_click(object sender, EventArgs e)
        {
            int i, j;
            Color pix1,pix2;
            for (i = 0; i < image.Height; i++)
            {
                for (j = 0; j < image.Width; j++)
                {
                    pix1 = image.GetPixel(j,i);
                    pix2 = Color.FromArgb(255-pix1.R , 255 -pix1.G, 255-pix1.B );
                    image.SetPixel(j, i, pix2);
                }
            }
            pictureBox1.Image = image;
        }
    }
